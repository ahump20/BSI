Real-time Updates

// Create real-time updating chart
const dataFetcher = async () => {
  const response = await fetch('/api/v1/live/scores');
  return await response.json();
};

const { chart, updater } = ChartFactory.createRealTime(
  'line',
  'liveCanvas',
  initialData,
  dataFetcher,
  { updateInterval: 2000 } // Update every 2 seconds
);

// Control updates
updater.stop();
updater.start(dataFetcher);
updater.setInterval(5000);

Chart Factory

const radarChart = ChartFactory.create('radar', 'canvas4', data);

// With custom options

  horizontal: true,
  stacked: true
});
